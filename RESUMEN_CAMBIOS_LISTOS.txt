     @app.post("/submit-visita")
     async def recibir_visita(request: Request):
         form_data = await request.form()
         
         # Capturar entrevistas
         entrevistas = []
         for i in range(1, 100):
             motivo = form_data.get(f"entrevista_{i}_motivo", "").strip()
             positivos = form_data.get(f"entrevista_{i}_positivos", "").strip()
             mejoras = form_data.get(f"entrevista_{i}_mejoras", "").strip()
             
             if motivo or positivos or mejoras:
                 entrevistas.append({
                     "numero_cliente": i,
                     "motivo_visita": motivo,
                     "aspectos_positivos": positivos,
                     "oportunidades_mejora": mejoras,
                 })
         
         # Guardar
         if entrevistas:
